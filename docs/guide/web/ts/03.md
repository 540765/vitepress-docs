## Boolean 类型

```ts
let isDone: boolean = false;
//ES5: var idDone = false;
```

## Number 类型

```ts
let count: number = 1;
//ES5:var count = 1;
```

## String 类型

```ts
let name: string = "chen";
//ES5:var name = "chen";
```

## Symbol 类型

```ts
const sym = Symbol();
let obj = {
  [sym]:"semlinker",
};

console.log(obj[sym]);// semlinker
```

## Array类型

```ts
let list:number[] = [1,2,3];
//ES5:var list = [1,2,3];

let list:Array<number> = [1,2,3];//泛型语法
//ES5:var list = [1,2,3];
```
